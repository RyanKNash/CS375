app.post("/rate", (req, res) => {
    const { recipe } = req.body;

    if (!Array.isArray(recipe)) {
        return res.status(400).json({});
    }

    const tooManyIngredients = recipe.length >= 3;
    return res.json({ tooManyIngredients });
});
